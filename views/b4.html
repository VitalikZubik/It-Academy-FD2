                function replaceСharacters (str) {
                    str = str.replace(/&/g, '&');
                    str = str.replace(/replace(/>/g, '>');
                    str = str.replace(/'/g, ''');
                    str = str.replace(/"/g, '"');
                
                    return str;
                }
                
                function buildWrapper(teg) {
                    return function (str) {
                        str = replaceСharacters(str);
                
                        return `<${teg}>${str}`
                    }
                }
                
                const wrapH1 = buildWrapper('H1');
